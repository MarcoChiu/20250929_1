題目三：計算球體體積

請製作一個球體積計算函式，參數為半徑，回傳計算後四捨五入取至第一位小數的結果。

球體積 = 4 / 3 x π x 球半徑³

// input 輸入
sphereVolume(1)
sphereVolume(10)
sphereVolume(20)

// output 輸出
4.2
4188.8
33510.3

回答:

        function sphereVolume(num) {
            if (!isNaN(num)) {//不是NaN時才運算     
                // 確保半徑是正數
                if (num < 0) {
                    console.log("半徑不能為負數");
                }
                // AI
                // 計算球體積：4/3 * π * r³
                let c = (4 / 3) * Math.PI * Math.pow(num, 3);//取小數點幾位   
                console.log( Number.parseFloat(c).toFixed(1));                
            } else {
                console.log(num + '-數字有問題無法運算');
            }
        }
        sphereVolume(1);
        sphereVolume(10);
        sphereVolume(20);

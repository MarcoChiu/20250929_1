題目一：多重條件

一週有七天，請製作一個函式可輸入不同的星期日數，並回傳當天的心情小語，如下例：
// input 輸入
myThought(1)
myThought(7)
myThought(8)

// output 輸出
"星期一：穿新衣。"
"星期日：來考試。"
"一周只有七天，輸入錯囉"

回答:

        function myThought(num) {
            if (!isNaN(num)) {//不是NaN時才運算            

                switch (num) {
                    case 1:
                        console.log("星期一穿新衣");
                        break;
                    case 2:
                        console.log("星期二肚子餓");
                        break;
                    case 3:
                        console.log("星期三去爬山");
                        break;
                    case 4:
                        console.log("星期四去考試");
                        break;
                    case 5:
                        console.log("星期五去跳舞");
                        break;
                    case 6:
                        console.log("星期六去斗六");
                        break;
                    case 7:
                        console.log("星期天去休息");
                        break;
                    default:
                        console.log(`一周只有七天，輸入錯囉`);
                }
            } else {
                console.log(num + '-數字有問題無法運算');
            }
        }

        // input 輸入
        myThought(1)
        myThought(7)
        myThought(8)
